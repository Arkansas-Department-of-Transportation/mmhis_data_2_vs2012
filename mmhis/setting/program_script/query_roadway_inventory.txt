SELECT [OBJECTID] [ID]
      ,CONVERT(VARCHAR(10), GETDATE(), 101) DateUpdated
      ,AH_District [District]
      ,AH_County [County]
      ,AH_Section [Section]
      ,AH_BLM [BeginningLogMile]
      ,AH_ELM [EndingLogMile]
      ,AH_Length [Length]
      ,[RuralUrbanArea] [RuralUrbanCode]
      ,[FunctionalClass] [FunctionalClassification]
      ,[NHS] [NHSSystemClassification]
      ,[SurfaceType] [SurfaceTypeCode]
      ,[SurfaceWidth]
      ,[LaneWidth]
      ,[MedianWidth]
      ,'S' + LTRIM(STR([RightShoulderSurface], 10)) [RightShoulderSurfaceType]
      ,'S' + LTRIM(STR([LeftShoulderSurface], 10)) [LeftShoulderSurfaceType]
      ,[RightShoulderWidth] [RightShoulderWidth]
      ,[LeftShoulderWidth]
      ,'' [JobNumber]
      ,[YearBuilt]
      ,[ADT] [AverageDailyTraffic]
      ,[ADT_Year] [YearADTCollected]
      ,[Total_Lanes] [NumberOfLanes]
  FROM [[[database]]].[[[schema]]].[[[table_name]]]
  WHERE AH_RoadID = '[[road_id]]'
  ORDER BY AH_BLM;
